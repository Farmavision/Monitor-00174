// Funzione per sbloccare l'orologio immediatamente function updateClock() { const ora = new Date(); document.getElementById('clock').innerText = ora.toLocaleTimeString('it-IT'); } // Chiamata immediata per evitare lo 00:00:00 updateClock(); setInterval(updateClock, 1000); // Logica di analisi oraria "corazzata" (non crasha se la cella è vuota) function isAperta(cella) { if(!cella) return false; try { const adesso = new Date(); const minAttuali = adesso.getHours() * 60 + adesso.getMinutes(); // Se la cella contiene "H24" o "APERTO" if(cella.toUpperCase().includes("H24")) return true; const fasce = cella.split(','); for(let f of fasce) { const ore = f.trim().split('-'); if(ore.length === 2) { const inP = ore[0].split(':'); const fiP = ore[1].split(':'); const mIn = parseInt(inP[0])*60 + (parseInt(inP[1]) || 0); const mFi = parseInt(fiP[0])*60 + (parseInt(fiP[1]) || 0); if(minAttuali >= mIn && minAttuali <= mFi) return true; } } } catch(e) { console.log("Errore cella:", cella); } return false; }
